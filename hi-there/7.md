# Ruby-like syntax in Swift

```ruby
5.times { puts 'We *love* Ruby -- it\'s outrageous!' }
```

```swift
for i in 0..5 {
  println("We *love* Swift -- it's outrageous!")
}
```

```swift
extension Int {
  func times(block: () -> ()) -> Int {
    for _ in 0..self {
      block()
    }
    return self
  }
}
```

```swift
extension Int {
  func times(block: (Int) -> ()) -> Int {
    for i in 0..self {
      block(i)
    }
    return self
  }
}
```

```swift
5.times { (i: Int) in println("\(i)") }
```
